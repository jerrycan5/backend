class Car { constructor(brand, model, year) { this.brand = brand; this.model = model; this.year = year; } displayinfo() { return `This ${this.brand} ${this.model} was made in the year ${this.year}.`; } } // Use correct variable naming const myCar = new Car("Toyota", "Corolla", 2022); // Correct function call and spacing document.getElementById("myCar").innerHTML = myCar.displayinfo();
